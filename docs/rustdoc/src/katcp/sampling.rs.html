9/// Sampling strategy types.
10#[derive(Debug, Clone, Copy, PartialEq)]
11pub enum SamplingStrategy {
12    /// No sampling — sensor updates are not reported.
13    None,
14    /// Automatic sampling (server chooses strategy).
15    Auto,
16    /// Sample at fixed period (seconds).
17    Period(u64, u32), // (secs, nanos)
18    /// Sample on every change.
19    Event,
20    /// Sample when value changes by at least the given difference.
21    Differential(f64),
22    /// Rate-limited event sampling: (min_secs, min_nanos, max_secs, max_nanos).
23    EventRate(u64, u32, u64, u32),
24    /// Rate-limited differential sampling: (threshold, min_secs, min_nanos, max_secs, max_nanos).
25    DifferentialRate(f64, u64, u32, u64, u32),
26}
27
28impl SamplingStrategy {
29    /// Get the string representation of the sampling strategy.
30    pub fn as_str(&self) -> &'static str {
31        match self {
32            SamplingStrategy::None => "none",
33            SamplingStrategy::Auto => "auto",
34            SamplingStrategy::Period(_, _) => "period",
35            SamplingStrategy::Event => "event",
36            SamplingStrategy::Differential(_) => "differential",
37            SamplingStrategy::EventRate(_, _, _, _) => "event-rate",
38            SamplingStrategy::DifferentialRate(_, _, _, _, _) => "differential-rate",
39        }
40    }
41
42    /// Parse a sampling strategy from its name and parameters.
43    ///
44    /// # Arguments
45    ///
46    /// * `name` - The strategy name (e.g., "event", "period").
47    /// * `params` - A slice of string parameters for the strategy.
48    ///
49    /// # Returns
50    ///
51    /// `Some(SamplingStrategy)` if the strategy and parameters are valid, or `None` otherwise.
52    pub fn parse(name: &str, params: &[&str]) -> Option<Self> {
53        match name {
54            "none" => Some(SamplingStrategy::None),
55            "auto" => Some(SamplingStrategy::Auto),
56            "event" => Some(SamplingStrategy::Event),
57            "period" => {
58                let secs: f64 = params.first()?.parse().ok()?;
59                let whole = secs as u64;
60                let frac = ((secs - whole as f64) * 1_000_000_000.0) as u32;
61                Some(SamplingStrategy::Period(whole, frac))
62            }
63            "differential" => {
64                let diff: f64 = params.first()?.parse().ok()?;
65                Some(SamplingStrategy::Differential(diff))
66            }
67            "event-rate" => {
68                let min_secs: f64 = params.first()?.parse().ok()?;
69                let min_whole = min_secs as u64;
70                let min_frac = ((min_secs - min_whole as f64) * 1_000_000_000.0) as u32;
71                let max_secs: f64 = params.get(1)?.parse().ok()?;
72                let max_whole = max_secs as u64;
73                let max_frac = ((max_secs - max_whole as f64) * 1_000_000_000.0) as u32;
74                Some(SamplingStrategy::EventRate(min_whole, min_frac, max_whole, max_frac))
75            }
76            "differential-rate" => {
77                let diff: f64 = params.first()?.parse().ok()?;
78                let min_secs: f64 = params.get(1)?.parse().ok()?;
79                let min_whole = min_secs as u64;
80                let min_frac = ((min_secs - min_whole as f64) * 1_000_000_000.0) as u32;
81                let max_secs: f64 = params.get(2)?.parse().ok()?;
82                let max_whole = max_secs as u64;
83                let max_frac = ((max_secs - max_whole as f64) * 1_000_000_000.0) as u32;
84                Some(SamplingStrategy::DifferentialRate(diff, min_whole, min_frac, max_whole, max_frac))
85            }
86            _ => Option::None,
87        }
88    }
89
90    /// Get the formatted parameters for this strategy.
91    ///
92    /// # Returns
93    ///
94    /// A space-separated string of parameters as used in the `#sensor-status` reply.
95    pub fn params_str(&self) -> String {
96        match self {
97            SamplingStrategy::None | SamplingStrategy::Auto | SamplingStrategy::Event => {
98                String::new()
99            }
100            SamplingStrategy::Period(s, n) => {
101                format!("{:.6}", *s as f64 + *n as f64 / 1_000_000_000.0)
102            }
103            SamplingStrategy::Differential(d) => format!("{}", d),
104            SamplingStrategy::EventRate(ms, mn, xs, xn) => {
105                format!(
106                    "{:.6} {:.6}",
107                    *ms as f64 + *mn as f64 / 1_000_000_000.0,
108                    *xs as f64 + *xn as f64 / 1_000_000_000.0
109                )
110            }
111            SamplingStrategy::DifferentialRate(d, ms, mn, xs, xn) => {
112                format!(
113                    "{} {:.6} {:.6}",
114                    d,
115                    *ms as f64 + *mn as f64 / 1_000_000_000.0,
116                    *xs as f64 + *xn as f64 / 1_000_000_000.0
117                )
118            }
119        }
120    }
121}

123/// An active sensor sampling observer that sends inform messages through a channel.
124pub struct SamplingObserver {
125    sensor_name: String,
126    strategy: SamplingStrategy,
127    tx: mpsc::UnboundedSender<Vec<u8>>,
128    state: Mutex<SamplingState>,
129}
130
131struct SamplingState {
132    last_sent: Option<Instant>,
133    last_value: Option<Vec<u8>>,
134}
135
136impl SamplingObserver {
137    /// Create a new sampling observer.
138    ///
139    /// # Arguments
140    ///
141    /// * `sensor_name` - The name of the sensor being observed.
142    /// * `strategy` - The sampling strategy to enforce.
143    /// * `tx` - The outgoing channel to send generated inform messages to.
144    ///
145    /// # Returns
146    ///
147    /// A new `SamplingObserver` wrapped in an `Arc`.
148    pub fn new(
149        sensor_name: String,
150        strategy: SamplingStrategy,
151        tx: mpsc::UnboundedSender<Vec<u8>>,
152    ) -> Arc<Self> {
153        Arc::new(Self {
154            sensor_name,
155            strategy,
156            tx,
157            state: Mutex::new(SamplingState {
158                last_sent: Option::None,
159                last_value: Option::None,
160            }),
161        })
162    }
163
164    /// Determine whether a sensor reading should be sent based on the active strategy.
165    fn should_send(&self, reading: &SensorReading) -> bool {
166        let mut state = self.state.lock().unwrap();
167        match self.strategy {
168            SamplingStrategy::None => false,
169            SamplingStrategy::Auto | SamplingStrategy::Event => {
170                let changed = state.last_value.as_ref() != Some(&reading.value);
171                if changed {
172                    state.last_value = Some(reading.value.clone());
173                    state.last_sent = Some(Instant::now());
174                }
175                changed
176            }
177            SamplingStrategy::Period(secs, nanos) => {
178                let period = std::time::Duration::new(secs, nanos);
179                let now = Instant::now();
180                let should = state
181                    .last_sent
182                    .map(|t| now.duration_since(t) >= period)
183                    .unwrap_or(true);
184                if should {
185                    state.last_sent = Some(now);
186                    state.last_value = Some(reading.value.clone());
187                }
188                should
189            }
190            SamplingStrategy::Differential(threshold) => {
191                let changed = if let Some(ref last) = state.last_value {
192                    let last_val: f64 = std::str::from_utf8(last)
193                        .unwrap_or("0")
194                        .parse()
195                        .unwrap_or(0.0);
196                    let cur_val: f64 = std::str::from_utf8(&reading.value)
197                        .unwrap_or("0")
198                        .parse()
199                        .unwrap_or(0.0);
200                    (cur_val - last_val).abs() >= threshold
201                } else {
202                    true
203                };
204                if changed {
205                    state.last_value = Some(reading.value.clone());
206                    state.last_sent = Some(Instant::now());
207                }
208                changed
209            }
210            SamplingStrategy::EventRate(min_s, min_n, _max_s, _max_n) => {
211                let min_period = std::time::Duration::new(min_s, min_n);
212                let now = Instant::now();
213                let changed = state.last_value.as_ref() != Some(&reading.value);
214                let rate_ok = state
215                    .last_sent
216                    .map(|t| now.duration_since(t) >= min_period)
217                    .unwrap_or(true);
218                if changed && rate_ok {
219                    state.last_value = Some(reading.value.clone());
220                    state.last_sent = Some(now);
221                    true
222                } else {
223                    false
224                }
225            }
226            SamplingStrategy::DifferentialRate(threshold, min_s, min_n, _max_s, _max_n) => {
227                let min_period = std::time::Duration::new(min_s, min_n);
228                let now = Instant::now();
229                let diff_ok = if let Some(ref last) = state.last_value {
230                    let last_val: f64 = std::str::from_utf8(last)
231                        .unwrap_or("0")
232                        .parse()
233                        .unwrap_or(0.0);
234                    let cur_val: f64 = std::str::from_utf8(&reading.value)
235                        .unwrap_or("0")
236                        .parse()
237                        .unwrap_or(0.0);
238                    (cur_val - last_val).abs() >= threshold
239                } else {
240                    true
241                };
242                let rate_ok = state
243                    .last_sent
244                    .map(|t| now.duration_since(t) >= min_period)
245                    .unwrap_or(true);
246                if diff_ok && rate_ok {
247                    state.last_value = Some(reading.value.clone());
248                    state.last_sent = Some(now);
249                    true
250                } else {
251                    false
252                }
253            }
254        }
255    }
256
257    /// Format and send the sensor reading as a `#sensor-status` inform message.
258    fn send_inform(&self, reading: &SensorReading) {
259        let inform = Message::inform(
260            "sensor-status",
261            vec![
262                format!("{:.6}", reading.timestamp).into_bytes(),
263                b"1".to_vec(), // count
264                self.sensor_name.as_bytes().to_vec(),
265                reading.status.as_str().as_bytes().to_vec(),
266                reading.value.clone(),
267            ],
268        );
269        let _ = self.tx.send(inform.to_bytes());
270    }
271}
272
273impl SensorObserver for SamplingObserver {
274    fn on_update(&self, _sensor_name: &str, reading: &SensorReading) {
275        if self.should_send(reading) {
276            self.send_inform(reading);
277        }
278    }

286    fn test_parse_strategy() {
287        assert_eq!(
288            SamplingStrategy::parse("none", &[]),
289            Some(SamplingStrategy::None)
290        );
291        assert_eq!(
292            SamplingStrategy::parse("auto", &[]),
293            Some(SamplingStrategy::Auto)
294        );
295        assert_eq!(
296            SamplingStrategy::parse("event", &[]),
297            Some(SamplingStrategy::Event)
298        );
299
300        match SamplingStrategy::parse("period", &["1.5"]) {
301            Some(SamplingStrategy::Period(1, n)) => assert!(n > 0),
302            other => panic!("expected Period, got {:?}", other),
303        }
304
305        assert_eq!(
306            SamplingStrategy::parse("differential", &["0.5"]),
307            Some(SamplingStrategy::Differential(0.5))
308        );

327        let (tx, mut rx) = mpsc::unbounded_channel();
328        let obs = SamplingObserver::new("test-sensor".into(), SamplingStrategy::Event, tx);
329
330        // First update should always send
331        let r1 = SensorReading::new(1.0, crate::sensor::SensorStatus::Nominal, b"10".to_vec());
332        obs.on_update("test-sensor", &r1);
333        assert!(rx.try_recv().is_ok());
334
335        // Same value should not send
336        let r2 = SensorReading::new(2.0, crate::sensor::SensorStatus::Nominal, b"10".to_vec());
337        obs.on_update("test-sensor", &r2);
338        assert!(rx.try_recv().is_err());
339
340        // Different value should send
341        let r3 = SensorReading::new(3.0, crate::sensor::SensorStatus::Nominal, b"20".to_vec());
